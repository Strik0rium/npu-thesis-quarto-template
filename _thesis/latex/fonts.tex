% 宋体/黑体：各 OS 系统字体；正文 \textbf 与标题加粗均走 BoldFont 映射。
% Scale=1.0 关闭 fontspec 对黑体的 MatchLowercase 自动缩小。
\IfFontExistsTF{Songti SC}{%
  \setCJKmainfont{Songti SC Light}[
    BoldFont = {Heiti SC Medium},
    BoldFeatures = { Scale = 1.0 },
    ItalicFont = {Songti SC},
    BoldItalicFont = {Heiti SC Medium},
    BoldItalicFeatures = { Scale = 1.0 },
  ]%
}{%
  \IfFontExistsTF{SimSun}{%
    \setCJKmainfont{SimSun}[
      BoldFont = SimHei,
      BoldFeatures = { Scale = 1.0 },
      ItalicFont = SimSun,
      BoldItalicFont = SimHei,
      BoldItalicFeatures = { Scale = 1.0 },
    ]%
  }{%
    \IfFontExistsTF{Noto Serif CJK SC}{%
      \setCJKmainfont{Noto Serif CJK SC}[
        BoldFont = {Noto Sans CJK SC Medium},
        BoldFeatures = { Scale = 1.0 },
        ItalicFont = {Noto Serif CJK SC},
        BoldItalicFont = {Noto Sans CJK SC Medium},
        BoldItalicFeatures = { Scale = 1.0 },
      ]%
    }{%
      \IfFontExistsTF{Source Han Serif SC}{%
        \setCJKmainfont{Source Han Serif SC}[
          BoldFont = {Source Han Sans SC Medium},
          BoldFeatures = { Scale = 1.0 },
          ItalicFont = {Source Han Serif SC},
          BoldItalicFont = {Source Han Sans SC Medium},
          BoldItalicFeatures = { Scale = 1.0 },
        ]%
      }{%
        \PackageWarning{thesis}{No suitable CJK fonts found.}%
      }%
    }%
  }%
}
